\newcommand\StyleTitleSimple{
  \renewcommand\@maketitle{%
  \newpage\null\vskip 2em%
  \begin{center}%
  \let\footnote\thanks
  {\LARGE \@title \par}\vskip 0.5em%
  {\large\@author, \today}%
  \end{center}%

  \vskip 1.5em
	}
}

  }{
  \vskip 0.5em
  }
}
